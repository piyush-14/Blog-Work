def mse_grad(out,tar):
  out.g = 2. * (out - tar.unsqueeze(1))/out.shape[0]

def relu_grad(inp,out):
  inp.g = (inp>0).float() * out.g

def lin_grad(inp, out, w, b):
  inp.g = out.g @ w.t()
  w.g = (inp * out.g).sum()
